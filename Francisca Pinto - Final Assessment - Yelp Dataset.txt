       user.fans

GROUP BY review.user_id
ORDER BY count_review_user DESC
;

---- User reviews and fans, ordered by the number of fans ----
SELECT review.user_id,
       COUNT(review.user_id) AS 'count_review_user',
       user.fans
FROM review LEFT JOIN user
ON review.user_id=user.id
GROUP BY review.user_id
HAVING user.fans>0
ORDER BY user.fans DESC
;
